  select="('AB','c', 'dEF', 'ghij', 'KLmno','PqrSt','uvwxy','z','123',
'456789')"/>

<xsl:for-each select="$rString">
<xsl:sort select="."  order="descending"   case-order ="upper-first"/>
  <xsl:value-of select="." separator="| "/>
