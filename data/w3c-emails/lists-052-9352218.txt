>> <owl:Class rdf:about="&eg;Bag"/>
>> <owl:ObjectProperty rdf:about="&eg:_1"/>
>> <owl:ObjectProperty rdf:about="&eg:_2"/>
>> <eg:Bag>
>>   <eg:1>
>>     <owl:Thing/>
>>   </eg:1>
>> </eg:Bag>

>>
>> <owl:Class rdf:about="&rdf;Bag"/>

>>   <rdf:1>
>>     <owl:Thing/>
>>   </rdf:1>
>> </rdf:Bag>
>>
