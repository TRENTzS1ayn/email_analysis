     earl:Thing a rdfs:Class .

[[[
earl:WebContent rdfs:domain earl:TestSubject; rdfs:subClassOf
earl:Thing .

     earl:WebContent rdfs:subClassOf earl:Thing .
     earl:Tool rdfs:subClassOf earl:Thing .
     earl:subjectType a rdf:Property;
        rdfs:domain earl:TestSubject; rdfs:range earl:Thing .
     earl:TestSubject a rdfs:Class .
