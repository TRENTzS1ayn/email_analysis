$ python bitest.py

 { ?x <#notEqualTo> ?z . } .
<#p> <#q> <#r> .
[[[<#p>, <#notEqualTo>, <#r>]]]

{ ?x ?y ?z .
  ?x <http://www.w3.org/2000/10/swap/log#uri> ?p . } => { ?x <#uri> ?p
. } .
<#p> <#q> <#r> .
<#x> <#y> <#z> .
[[[<#p>, <#uri>, "#p"]], [[<#x>, <#uri>, "#x"]]]

{ ?x ?y ?z .
 ?x <http://www.w3.org/2000/10/swap/string#notEqualTo> ?z . } =>
 { ?x <#notEqualTo> ?z . } .
"p" <#q> "r" .
<#p> <#q> <#p> .
"p" <#q> "p" .
"x" _:y "z" .
[[["p", <#notEqualTo>, "r"]], [["x", <#notEqualTo>, "z"]]]
